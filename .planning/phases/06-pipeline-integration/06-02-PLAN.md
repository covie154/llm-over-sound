    ```python
    """Snapshot golden-file tests for pipeline rendering (D-25, D-26).

    Golden files in tests/snapshots/ are manually authored. Tests fail
    if rendered output doesn't match. Whitespace is normalised before
    comparison (stripped, Unix line endings) per Pitfall 5 in RESEARCH.md.
    """
    import json
    import pathlib
    import pytest

    SNAPSHOTS_DIR = pathlib.Path(__file__).parent / "snapshots"

    def _normalize(text: str) -> str:
        """Normalize whitespace for snapshot comparison."""
        return text.strip().replace("\r\n", "\n")

    def _render_via_pipeline(llm_pipeline, study_type: str, findings: dict,
                             technique: dict | None = None) -> str:
        """Helper: render through pipeline and return ct."""
        payload = {"study_type": study_type, "findings": findings}
        if technique:
            payload["technique"] = technique
        ct = json.dumps(payload)
        result = llm_pipeline.process({"id": "snap", "fn": "render", "ct": ct})
        assert result["st"] == "S", f"Render failed: {result['ct']}"
        return result["ct"]

    def _load_golden(name: str) -> str:
        """Load and normalize a golden snapshot file."""
        path = SNAPSHOTS_DIR / name
        return _normalize(path.read_text(encoding="utf-8"))

    # ===== Snapshot Tests =====

    def test_snapshot_us_hbs(llm_pipeline):
        findings = {
            "liver": "Normal in size and echotexture. No focal lesion.",
            "gallbladder_and_cbd": "Gallbladder is well distended with no calculi. CBD measures 4mm.",
            "spleen": "Normal in size.",
            "pancreas": "Visualised portions are unremarkable.",
        }
        technique = {"cbd_diameter": "4"}
        rendered = _render_via_pipeline(llm_pipeline, "us hbs", findings, technique)
        assert _normalize(rendered) == _load_golden("us_hbs_render.txt")

    def test_snapshot_ct_ap_freeform(llm_pipeline):
        findings = {
            "liver": "Normal in size and attenuation. No focal lesion.",
            "spleen_adrenals_pancreas": "Unremarkable.",
            "kidneys_and_ureters": "No hydronephrosis or calculi.",
        }
        rendered = _render_via_pipeline(llm_pipeline, "ct ap", findings)
        assert _normalize(rendered) == _load_golden("ct_ap_freeform_render.txt")

    def test_snapshot_ct_ap_structured(llm_pipeline):
        findings = {
            "liver": "Normal in size and attenuation. No focal lesion.",
            "spleen_adrenals_pancreas": "Unremarkable.",
            "kidneys_and_ureters": "No hydronephrosis or calculi.",
        }
        rendered = _render_via_pipeline(llm_pipeline, "ct ap structured", findings)
        assert _normalize(rendered) == _load_golden("ct_ap_structured_render.txt")

    def test_snapshot_ct_tap(llm_pipeline):
        findings = {
            "lungs": "Clear. No focal consolidation.",
            "pleura": "No effusion.",
            "liver": "Normal in size and attenuation.",
            "spleen_adrenals_pancreas": "Unremarkable.",
        }
        rendered = _render_via_pipeline(llm_pipeline, "ct tap", findings)
        assert _normalize(rendered) == _load_golden("ct_tap_render.txt")
    ```

    **Important:** The exact findings used in tests must match the findings used to generate the golden files. The golden files must be authored from the actual render output, then committed.

    **Approach for creating golden files:** Run each render call via a one-off Python script, capture output, save to snapshots dir. Review the output for correctness. Commit as the golden reference. Do NOT auto-update golden files in the test itself.

    Then create `render_demo.py`:

    ```python
    #!/usr/bin/env python3
    """Minimal render demo -- standalone pipeline usage without ggwave.

    Demonstrates that LLMPipeline is importable and callable independently
    of the backend audio loop (per D-18, D-27).

    Usage:
        cd python-backend
        python examples/render_demo.py
    """

    import json
    import sys
    import os

    # Ensure python-backend is on the path when running from examples/
    sys.path.insert(0, os.path.join(os.path.dirname(__file__), ".."))

    from lib.pipeline import LLMPipeline


    def main():
        # Instantiate pipeline (loads all templates from rpt_templates/)
        pipeline = LLMPipeline()

        # Prepare a US HBS render request
        payload = json.dumps({
            "study_type": "us hbs",
            "findings": {
                "liver": "Normal in size and echotexture. No focal lesion.",
                "gallbladder_and_cbd": (
                    "Gallbladder is well distended with no calculi. "
                    "CBD measures 4mm."
                ),
                "spleen": "Normal in size.",
                "pancreas": "Visualised portions are unremarkable.",
            },
            "technique": {"cbd_diameter": "4"},
            "rest_normal": False,
        })

        msg = {"id": "demo01", "fn": "render", "ct": payload}

        print("=" * 60)
        print("LLM Pipeline Render Demo")
        print("=" * 60)
        print(f"\nStudy type: us hbs")
        print(f"Function: render (bypasses LLM stages)\n")

        result = pipeline.process(msg)

        print(f"Status: {result['st']}")
        print(f"\n{'=' * 60}")
        print("RENDERED REPORT:")
        print("=" * 60)
        print(result["ct"])


    if __name__ == "__main__":
        main()
    ```

    The script must be runnable with `cd python-backend && python examples/render_demo.py` and print a rendered US HBS report to stdout.
